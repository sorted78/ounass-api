🔴 PROBLEM: Pandas won't install on Python 3.13

✅ SOLUTION: Use Python 3.12 (Recommended)

Step 1: Install Python 3.12
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
brew install python@3.12

Step 2: Create Virtual Environment with Python 3.12
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
cd /Users/root1/Desktop/Projects/ounass-api
/opt/homebrew/opt/python@3.12/bin/python3.12 -m venv venv
source venv/bin/activate

Step 3: Install Dependencies with Stable Versions
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
pip install --upgrade pip
pip install -r requirements-py39-py312.txt

Step 4: Verify Installation
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
python -c "import pandas, numpy, sklearn, fastapi; print('✅ All packages OK!')"

Step 5: Run the API
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
uvicorn src.main:app --reload
